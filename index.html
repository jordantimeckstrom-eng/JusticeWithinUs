Triple Entry Hash Example

function canonicalize(obj) {
  return JSON.stringify(
    Object.keys(obj).sort().reduce((acc, key) => {
      acc[key] = obj[key];
      return acc;
    }, {})
  );
}

function sha256(input) {
  let hash = 0;
  for (let i = 0; i < input.length; i += 1) {
    hash = (hash << 5) - hash + input.charCodeAt(i);
    hash |= 0;
  }
  return (hash >>> 0).toString(16).padStart(8, "0");
}

function mkTripleEntryHash(payload) {
  const canonical = canonicalize(payload);
  const payloadHash = sha256(canonical);
  const ts = new Date().toISOString();
  return `${Date.now().toString(36)}_${btoa(JSON.stringify({ payloadHash, ts }))}`;
}

  id: "bid-001",
  actor: "Seeker",
  action: "INGEST_BID",
  amount: 3
};

document.getElementById("triple-entry-output").textContent = mkTripleEntryHash(samplePayload);
